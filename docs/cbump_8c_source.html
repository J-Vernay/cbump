20  (move.x > 0) ? (other->x - box->x - box->w) : (other->x + other->w - box->x),
21  (move.y > 0) ? (other->y - box->y - box->h) : (other->y + other->h - box->y),

33 cbump_collision cbump_move_simple(cbump_box* box, cbump_vec2 move,
34  unsigned nb_obstacles, cbump_box* obstacles[])
35 {
36  // Keeping track of the nearest collision.
37  cbump_collision collision = {box, NULL, NULL, {0,0}, {0,0}, 1};
38 
39  for (unsigned i = 0; i < nb_obstacles; ++i) {
40  cbump_box* obstacle = obstacles[i];
41  // For each obstacle, determine its time of impact on both axis.
42  cbump_vec2 toi = cbump_time_of_impact(box, obstacle, move);
43  // Check if impact position would actually create collision.
44  bool collide_x = toi.x < collision.time
45  && _range_overlap(box->y + toi.x*move.y, box->h, obstacle->y, obstacle->h);
46  bool collide_y = toi.y < collision.time
47  && _range_overlap(box->x + toi.y*move.x, box->w, obstacle->x, obstacle->w);
48 
49  if (!collide_x && !collide_y)
50  continue;
51 
52  // Update collision to current obstacle.
53  collision.index = i;
54  if (toi.x < toi.y) {
55  collision.time = toi.x;
56  collision.normal.x = move.x > 0 ? -1 : 1;
57  collision.normal.y = 0;
58  } else {
59  collision.time = toi.y;
60  collision.normal.x = 0;
61  collision.normal.y = move.y > 0 ? -1 : 1;
62  }
63  }
64  // Update box position.
65  box->x += collision.time * move.x;
66  box->y += collision.time * move.y;
67 
68  collision.obstacle = obstacles[collision.index];
69  collision.moveleft.x = (1 - collision.time) * move.x;
70  collision.moveleft.y = (1 - collision.time) * move.y;
71 
72  return collision;
73 }

113  unsigned nb_obstacles, cbump_box* obstacles[],
